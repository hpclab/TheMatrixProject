
FILEA (FileInputModule)
parameters
      inputFileName = EXE_TEST.csv
      inputSchema   = IADexe
end

FILEB (FileInputModule)
parameters
      inputFileName = PERSON_TEST.csv
      inputSchema   = IADperson
end



PRODUCT (ProductModule)
inputs
	IADexe = FILEA
	IADperson = FILEB
parameters
	IDfield = PERSON_ID
	functions = [lessThan(FILEA.PERSON_ID,"11")]
end


FILEOUTPUT_PRODUCT (FileOutputModule)
inputs 
	custom = PRODUCT
parameters
	checksum = none
	compression = none
end
