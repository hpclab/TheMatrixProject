
OUTPATFile (FileInputModule)
parameters
	  inputFileName = oBIG.csv
      inputSchema   = IADoutpat
end

OUTPAT_Sorted (SortModule)
inputs
  IADoutpat = OUTPATFile
parameters
  fieldNames = [PERSON_ID]
end

AggregateOUTPAT (AggregateModule)
inputs
	IADoutpat = OUTPAT_Sorted
parameters
	isInputSorted = true
	groupBy =  [ PERSON_ID ]
	functions = [ COUNT(PERSON_ID) ]
	results =  [ { NUMBER_OF_PROC : int } ]
end
