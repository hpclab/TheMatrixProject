HOSP (ScriptInputModule)
parameters
	scriptFilename = LoadHOSP.txt
	inputName = HOSP
	expectedSchema = IADhosp
end

// Generate last Year
ClLastYearHOSP (ExtendDataModule)
inputs
	IADhosp = HOSP
parameters
  attributes = [ { LastYear: boolean } ]
end

LastYearHOSP (ApplyFunction)
inputs
	custom = ClLastYearHOSP
parameters
	function = elapsedTimeInRange(START_DATE, 2011-12-31, 0, 366 )
	result   = LastYear
end

// Filter last year in HOSP
FilterLastYearHOSP (FilterModule)
inputs
	custom = LastYearHOSP
parameters
	conditions = [{LastYear=true}]
	
end
